Create `.github/workflows/scrape-cards.yml`:

```yaml
name: Scrape STS2 Card Data

on:
  schedule:
    - cron: '0 6 * * 1'   # Every Monday at 06:00 UTC (weekly cadence matches STS2 patch rhythm)
  workflow_dispatch:       # Manual trigger for testing and patch-day runs

permissions:
  contents: write          # Needed to commit cards.json

jobs:
  scrape:
    runs-on: ubuntu-latest
    timeout-minutes: 15

    steps:
      - name: Checkout repo
        uses: actions/checkout@v4

      - name: Set up Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '22'
          cache: 'npm'
          cache-dependency-path: scripts/scraper/package-lock.json

      - name: Install scraper dependencies
        working-directory: scripts/scraper
        run: npm ci

      - name: Install Playwright browsers
        working-directory: scripts/scraper
        run: npx playwright install chromium --with-deps

      - name: Run scraper
        working-directory: scripts/scraper
        run: npm run scrape
        # Non-zero exit here = scraper validation failed = job fails loudly
        # Never reaches the commit step if this fails

      - name: Check for changes
        id: diff
        run: |
          git diff --quiet public/cards.json && echo "changed=false" >> $GITHUB_OUTPUT || echo "changed=true" >> $GITHUB_OUTPUT

      - name: Commit updated cards.json
        if: steps.diff.outputs.changed == 'true'
        run: |
          git config user.name "github-actions[bot]"
          git config user.email "github-actions[bot]@users.noreply.github.com"
          git add public/cards.json
          CARD_COUNT=$(node -e "console.log(JSON.parse(require('fs').readFileSync('public/cards.json','utf8')).length)")
          git commit -m "data: update cards.json (${CARD_COUNT} cards) [skip ci]"
          git push

      - name: No changes detected
        if: steps.diff.outputs.changed == 'false'
        run: echo "cards.json is up to date — no commit needed"
```

Key decisions documented in the workflow comments:
- Weekly Monday 06:00 UTC: balances freshness (STS2 patches often drop mid-week) without hammering Mobalytics daily
- `workflow_dispatch` allows manual trigger on patch day without waiting for the cron
- `[skip ci]` in commit message prevents an infinite loop where the commit triggers another workflow run
- `timeout-minutes: 15` prevents runaway Playwright sessions from consuming CI minutes
- Playwright install uses `--with-deps` to install system dependencies required on Ubuntu (avoids "missing library" failures that are the #1 Playwright-on-CI failure mode)
- `npm ci` (not `npm install`) for reproducible installs in CI
